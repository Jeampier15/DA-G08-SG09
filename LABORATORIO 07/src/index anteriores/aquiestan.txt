import React from 'react';
import ReactDOM from "react-dom/client";
import { BrowserRouter, Routes, Route } from "react-router-dom";
import Diseño from './paginas/Diseño';
import Inicio from './paginas/Inicio';
import Acerca from './paginas/Acerca';
import Servicios from './paginas/Servicios';
import DesarrolloWeb from './paginas/DesarrolloWeb';
import DesarrolloMovil from './paginas/DesarrolloMovil';
import NoEncontrado from './paginas/NoEncontrado';

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<Diseño />}>
          <Route index element={<Inicio />} />
          <Route path="acerca" element={<Acerca />} />
          <Route path="servicios" element={<Servicios />}>
            <Route path="desarrollo-web" element={<DesarrolloWeb />} />
            <Route path="desarrollo-movil" element={<DesarrolloMovil />} />
          </Route>
          <Route path="*" element={<NoEncontrado />} />
        </Route>
      </Routes>
    </BrowserRouter>
  );
}

const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
